/*******************************************************************************************
 * Program: Bot.js
 * Description:
 *   Represents an AI-controlled player in the Battleship game.
 *   - Determines firing decisions based on difficulty levels (EASY, MEDIUM, HARD)
 *   - Supports random targeting and smart targeting strategies
 *   - Can analyze previous hits to infer ship orientation and continue attacks
 *   - Automatically places ships randomly on the board during setup
 *   - Supports a power-up mechanic to enhance gameplay
 *
 * Inputs:
 *   - Player instance controlled by the bot
 *   - Difficulty level that determines decision-making behavior
 *   - Board state information for targeting and ship placement
 *
 * Output:
 *   - Selected firing coordinates based on strategy
 *   - Automated ship placements on the board
 *   - Updated player power-up state when activated
 *
 * Collaborators:
 *   - Brendan Coughlan
 *   - Jiaxing Rong
 *
 * Creation Date: 2026-03-04
 * Revision Dates:
 *   - Initial implementation of AI targeting logic
 *******************************************************************************************/

/* =========================

========================= */

/**
 * Represents an AI bot player.
 * @class
 */
export class Bot
{
  /**
   * Creates a new Bot instance.
   *
   * @param {Player} player - The player instance controlled by this bot.
   * @param {"EASY"|"MEDIUM"|"HARD"} difficulty - AI difficulty level.
   */
  constructor(player, difficulty)
  {
    this.player = player;
    this.difficulty = difficulty;
  }

  /**
   * Determines a smart target based on existing hits.
   *
   * - If multiple hits are aligned, the bot infers ship direction
   * - Attempts to extend attack along that direction
   * - Falls back to adjacent probing if direction is unclear
   *
   * @param {Board} board - Opponent's board
   * @returns {Cell|null} A target cell or null if no smart move found
   */
  getSmartTarget(board)
  {
    const hits = [];

    // Collect all valid HIT cells (not sunk)
    for (let col = 0; col < board.boardSize; col++)
    {
      for (let row = 0; row < board.boardSize; row++)
      {
        const cell = board.cells[col][row];
        if (cell.state === "HIT" && !cell.ship.isSunk())
        {
          hits.push(cell);
        }
      }
    }

    if (hits.length === 0) return null;

    // Try to detect ship orientation
    if (hits.length >= 2)
    {
      for (let i = 0; i < hits.length; i++)
      {
        for (let j = i + 1; j < hits.length; j++)
        {
          const a = hits[i];
          const b = hits[j];

          // Horizontal
          if (a.row === b.row)
          {
            const row = a.row;
            const cols = hits
              .filter(c => c.row === row)
              .map(c => c.col)
              .sort((a, b) => a - b);

            const left = cols[0] - 1;
            const right = cols[cols.length - 1] + 1;

            if (left >= 0)
            {
              const cell = board.cells[left][row];
              if (cell.state === "EMPTY") return cell;
            }

            if (right < board.boardSize)
            {
              const cell = board.cells[right][row];
              if (cell.state === "EMPTY") return cell;
            }
          }

          // Vertical
          if (a.col === b.col)
          {
            const col = a.col;
            const rows = hits
              .filter(c => c.col === col)
              .map(c => c.row)
              .sort((a, b) => a - b);

            const up = rows[0] - 1;
            const down = rows[rows.length - 1] + 1;

            if (up >= 0)
            {
              const cell = board.cells[col][up];
              if (cell.state === "EMPTY") return cell;
            }

            if (down < board.boardSize)
            {
              const cell = board.cells[col][down];
              if (cell.state === "EMPTY") return cell;
            }
          }
        }
      }
    }

    // Fallback: probe neighbors
    return this.getNextBestCell(board, hits[0]);
  }

  /**
   * Places a ship randomly on the board.
   *
   * Continues attempting random placements until successful.
   *
   * @param {number} length - Length of the ship
   * @returns {void}
   */
  placeNextShip(length)
  {
    const directions = ["N", "E", "S", "W"];
    let placed = false;

    while (!placed)
    {
      const randomRow = Math.floor(Math.random() * this.player.board.boardSize);
      const randomCol = Math.floor(Math.random() * this.player.board.boardSize);

      const randomDirection =
        directions[Math.floor(Math.random() * directions.length)];

      this.player.setOrientation(randomDirection);

      const targetCell = this.player.board.cells[randomRow][randomCol];

      placed = this.player.placeShipAt(
        targetCell.x + 1,
        targetCell.y + 1,
        length,
      );
    }
  }

  /**
   * Returns a random valid cell to fire at.
   *
   * @param {Board} board - Opponent's board
   * @returns {Cell} A randomly selected empty cell
   */
  getRandomCell(board)
  {
    let randomCell = null;

    while (!randomCell)
    {
      const x = Math.floor(Math.random() * board.boardSize);
      const y = Math.floor(Math.random() * board.boardSize);

      if (board.cells[x][y].state === "EMPTY")
      {
        randomCell = board.cells[x][y];
      }
    }

    return randomCell;
  }

  /**
   * Finds the next best adjacent cell around a hit.
   *
   * @param {Board} board - Opponent's board
   * @param {Cell} cell - Starting hit cell
   * @returns {Cell|null} Next candidate cell
   */
  getNextBestCell(board, cell)
  {
    const directions = [
      [0, -1], // N
      [1, 0],  // E
      [0, 1],  // S
      [-1, 0], // W
    ];

    for (const [dx, dy] of directions)
    {
      const newCol = cell.col + dx;
      const newRow = cell.row + dy;

      if (
        newCol >= 0 &&
        newCol < board.boardSize &&
        newRow >= 0 &&
        newRow < board.boardSize
      )
      {
        const nextCell = board.cells[newCol][newRow];

        if (nextCell.state === "EMPTY")
        {
          return nextCell;
        }
      }
    }

    return null;
  }

  /**
   * Determines the firing location based on difficulty.
   *
   * @param {Board} opponentBoard - Opponent's board
   * @returns {{selectedX: number, selectedY: number}} Pixel coordinates
   */
  getFireLocation(opponentBoard)
  {
    let selectedCell = null;

    switch (this.difficulty)
    {
      case "EASY":
        selectedCell = this.getRandomCell(opponentBoard);
        break;

      case "MEDIUM":
        if (Math.random() <= 0.25) this.usePowerup();

        const smartCell = this.getSmartTarget(opponentBoard);
        selectedCell = smartCell || this.getRandomCell(opponentBoard);
        break;

      case "HARD":
        this.usePowerup();

        for (let col = 0; col < opponentBoard.boardSize; col++)
        {
          for (let row = 0; row < opponentBoard.boardSize; row++)
          {
            const cell = opponentBoard.cells[col][row];
            if (cell.ship && cell.state === "EMPTY")
            {
              selectedCell = cell;
              break;
            }
          }
        }
        break;
    }

    return {
      selectedX: selectedCell.x + 1,
      selectedY: selectedCell.y + 1,
    };
  }

  /**
   * Activates the bot's power-up ability.
   *
   * @returns {void}
   */
  usePowerup()
  {
    if (!this.player.has_powerup) return;

    this.player.powerup_active = true;
    this.player.has_powerup = false;
  }
}
